BluetoothSerial SerialBT;
int received;// received value will be stored in this variable
char receivedChar;// received value will be stored as CHAR in this variable
String readString;
const char turnON ='ON';
const char turnOFF ='OFF';
const int LEDpin = 2;

void setup() {
  Serial.begin(115200);
  SerialBT.begin("ESP32_DK`"); //Bluetooth device name
  Serial.println("The device started, now you can pair it with bluetooth!");
  Serial.println("To turn ON send: a");//print on serial monitor  
  Serial.println("To turn OFF send: b"); //print on serial monitor 
  pinMode(LEDpin, OUTPUT);
}

void loop() {
    

  while (SerialBT.available()) {
    receivedChar = SerialBT.read();
    readString += receivedChar;
    //Serial.print(readString);
  }
    //Serial.write(SerialBT.read());  
    //Serial.write(SerialBT.read()); 
    //Serial.write(SerialBT.read()); 
    //SerialBT.print("Received:");// write on BT app

    //Serial.write(receivedChar);

    //SerialBT.println(receivedChar);//print on the app    
    //SerialBT.write(receivedChar); //print on serial monitor

   if (readString.length() >0) {
    Serial.println(readString);
    if(readString == "ON")
    //if(receivedChar == turnON)
    {
     SerialBT.println("LED ON:");// write on BT app
     Serial.println("LED ON:");//write on serial monitor
     digitalWrite(LEDpin, HIGH);// turn the LED ON
       
    }
    if(readString == "OFF")
    //if(receivedChar == turnOFF)
    {
     SerialBT.println("LED OFF:");// write on BT app
     Serial.println("LED OFF:");//write on serial monitor
     digitalWrite(LEDpin, LOW);// turn the LED off 
    }

    readString="";    
  }
